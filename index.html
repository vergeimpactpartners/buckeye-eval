// ── ENTRY POINTS ───────────────────────────────────────────────────────────── function doGet(e) { var action = e && e.parameter && e.parameter.action ? e.parameter.action : ''; var result; try { if (action === 'load_publish') { result = loadPublish(); } else if (action === 'load_photovoice') { result = loadPhotovoice(e.parameter.topic || ''); } else if (action === 'fetch_doc') { result = fetchDoc(e.parameter.doc_id || ''); } else { result = { ok: false, error: 'Unknown GET action: ' + action }; } } catch (err) { result = { ok: false, error: err.message }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var body; try { body = JSON.parse(e.postData.contents); } catch (err) { return jsonResponse({ ok: false, error: 'Invalid JSON body' }); } var action = body.action || ''; var payload = body.payload || body; // platform sends payload as top-level keys too var result; try { if (action === 'save_publish') { result = savePublish(payload); } else if (action === 'save_fidelity') { result = saveFidelity(payload); } else if (action === 'save_photovoice') { result = savePhotovoice(payload); } else { result = { ok: false, error: 'Unknown POST action: ' + action }; } } catch (err) { result = { ok: false, error: err.message }; } return jsonResponse(result); } // ── HELPERS ─────────────────────────────────────────────────────────────────── function jsonResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } /** * Returns (or creates) a folder by name inside a parent folder. */ function getOrCreateFolder(parent, name) { var iter = parent.getFoldersByName(name); if (iter.hasNext()) return iter.next(); return parent.createFolder(name); } /** * Returns (or creates) the root VIPIlluminate folder in My Drive. */ function getRootFolder() { var root = DriveApp.getRootFolder(); return getOrCreateFolder(root, ROOT_FOLDER_NAME); } /** * Overwrites (or creates) a JSON file by name in a given folder. */ function saveJsonFile(folder, filename, data) { var json = JSON.stringify(data, null, 2); var iter = folder.getFilesByName(filename); if (iter.hasNext()) { var file = iter.next(); file.setContent(json); return file.getId(); } else { var blob = Utilities.newBlob(json, 'application/json', filename); var file = folder.createFile(blob); return file.getId(); } } /** * Reads a JSON file by name from a folder. Returns {} if not found. */ function readJsonFile(folder, filename) { var iter = folder.getFilesByName(filename); if (!iter.hasNext()) return {}; try { return JSON.parse(iter.next().getBlob().getDataAsString()); } catch (e) { return {}; } } /** * Returns (or creates) a Google Sheet by name in a given Drive folder. * Creates the sheet as a new Spreadsheet if it doesn't exist. */ function getOrCreateSheet(folder, sheetName) { var iter = folder.getFilesByName(sheetName); if (iter.hasNext()) { var ssFile = iter.next(); return SpreadsheetApp.openById(ssFile.getId()).getActiveSheet(); } var ss = SpreadsheetApp.create(sheetName); // Move to the VIPIlluminate folder var ssFile = DriveApp.getFileById(ss.getId()); folder.addFile(ssFile); DriveApp.getRootFolder().removeFile(ssFile); return ss.getActiveSheet(); } /** * Returns a compact UTC timestamp string safe for filenames: 20260624T183045Z */ function nowStamp() { return Utilities.formatDate(new Date(), 'UTC', "yyyyMMdd'T'HHmmss'Z'"); } // ── ACTION: load_publish ────────────────────────────────────────────────────── /** * Returns the saved publish state object (themes, active activity, meeting * data, up-next labels, etc.). Called via GET on page load. */ function loadPublish() { var folder = getRootFolder(); var data = readJsonFile(folder, 'publish_state.json'); return { ok: true, data: data }; } // ── ACTION: save_publish ────────────────────────────────────────────────────── /** * Merges incoming payload into the existing publish state and saves it. * The platform sends the full publishedWidgets object each time, so a * simple overwrite is safe — but we merge to be defensive. */ function savePublish(payload) { var folder = getRootFolder(); var existing = readJsonFile(folder, 'publish_state.json'); // Merge: incoming keys overwrite existing keys var merged = Object.assign({}, existing, payload); // Strip the action key if it leaked in delete merged.action; saveJsonFile(folder, 'publish_state.json', merged); return { ok: true, data: merged }; } // ── ACTION: save_fidelity ───────────────────────────────────────────────────── /** * Saves fidelity assessment data. */ function saveFidelity(payload) { var folder = getRootFolder(); delete payload.action; saveJsonFile(folder, 'fidelity_data.json', payload); return { ok: true }; } // ── ACTION: save_photovoice ─────────────────────────────────────────────────── /** * Saves a photovoice submission to a Google Sheet (one row per submission) * and uploads the media file to Drive. Returns the Drive file link. * * Sheet columns: Timestamp | Topic | Type | Theme(s) | Drive Link | Thumbnail URL * Sheet name: photovoice_submissions */ function savePhotovoice(payload) { var topic = (payload.topic || 'unknown').toLowerCase().replace(/[^a-z0-9]/g, '_'); var type = payload.type || 'unknown'; var code = (payload.code || 'NO-CODE').replace(/[^a-zA-Z0-9\-_,\s]/g, '_'); var ts = payload.timestamp || new Date().toISOString(); var root = getRootFolder(); var pvFolder = getOrCreateFolder(root, 'photovoice'); var topicFolder = getOrCreateFolder(pvFolder, topic); var driveLink = ''; var thumbUrl = ''; // Upload media file to Drive if (type === 'story' && payload.storyText) { var ext = 'txt'; var blob = Utilities.newBlob(payload.storyText, 'text/plain', code + '_story_' + nowStamp() + '.txt'); var file = topicFolder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); driveLink = 'https://drive.google.com/file/d/' + file.getId() + '/view'; } else if (payload.fileData && payload.mimeType) { try { var ext = (payload.filename || 'file').split('.').pop(); var decoded = Utilities.base64Decode(payload.fileData); var mediaBlob = Utilities.newBlob(decoded, payload.mimeType, code + '_' + type + '_' + nowStamp() + '.' + ext); var mediaFile = topicFolder.createFile(mediaBlob); mediaFile.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); driveLink = 'https://drive.google.com/file/d/' + mediaFile.getId() + '/view'; if (type === 'photo') { thumbUrl = 'https://drive.google.com/thumbnail?id=' + mediaFile.getId() + '&sz=w600'; } } catch(e) { driveLink = 'Upload error: ' + e.message; } } // Append row to spreadsheet var sheet = getOrCreateSheet(root, 'photovoice_submissions'); // Add header row if sheet is empty if (sheet.getLastRow() === 0) { sheet.appendRow(['Timestamp', 'Topic', 'Type', 'Theme(s)', 'Description', 'Drive Link', 'Thumbnail URL']); sheet.getRange(1, 1, 1, 7).setFontWeight('bold'); } sheet.appendRow([ts, payload.topic, type, code, payload.description || '', driveLink, thumbUrl]); return { ok: true, driveLink: driveLink, thumbUrl: thumbUrl }; } // ── ACTION: load_photovoice ─────────────────────────────────────────────────── /** * Reads all submissions for a topic from the photovoice_submissions sheet. * Returns array of submission objects with driveLink and thumbnailUrl. */ function loadPhotovoice(topic) { if (!topic) return { ok: false, error: 'topic required' }; var root = getRootFolder(); var sheet = getOrCreateSheet(root, 'photovoice_submissions'); if (sheet.getLastRow() <= 1) return { ok: true, topic: topic, submissions: [] }; var data = sheet.getDataRange().getValues(); var headers = data[0]; // Timestamp, Topic, Type, Theme(s), Description, Drive Link, Thumbnail URL var submissions = []; for (var i = 1; i < data.length; i++) { var row = data[i]; var rowTopic = String(row[1] || '').toLowerCase().replace(/[^a-z0-9]/g, '_'); var filterTopic = topic.toLowerCase().replace(/[^a-z0-9]/g, '_'); if (rowTopic !== filterTopic) continue; submissions.push({ timestamp: String(row[0]), topic: String(row[1]), type: String(row[2]), code: String(row[3]), description: String(row[4]), driveLink: String(row[5]), thumbUrl: String(row[6]), files: row[5] ? [{ viewUrl: String(row[5]), thumbnailUrl: String(row[6]) }] : [] }); } // Newest first submissions.reverse(); return { ok: true, topic: topic, submissions: submissions }; } // ── ACTION: fetch_doc ───────────────────────────────────────────────────────── /** * Fetches the plain text content of a Google Doc by ID or URL. * The file must be accessible by the script's owner account. * * Pass: ?action=fetch_doc&doc_id= */ function fetchDoc(docIdOrUrl) { if (!docIdOrUrl) return { ok: false, error: 'doc_id required' }; // Extract ID from URL if full URL passed var docId = docIdOrUrl; var urlMatch = docIdOrUrl.match(/\/d\/([a-zA-Z0-9_-]{25,})/); if (urlMatch) docId = urlMatch[1]; try { var doc = DocumentApp.openById(docId); var body = doc.getBody(); var text = body.getText(); var title = doc.getName(); return { ok: true, title: title, text: text, charCount: text.length }; } catch(e) { return { ok: false, error: 'Could not open document: ' + e.message + '. Make sure the document is shared with ' + Session.getActiveUser().getEmail() }; } }
